# 📄 GitHub Actions Workflow للنسخ اليدوي
# انسخ المحتوى التالي إلى .github/workflows/nodejs-ci.yml عبر موقع GitHub

name: Node.js CI

on:
  push:
    branches: [ "main" ]
  pull_request:
    branches: [ "main" ]

jobs:
  build-and-test:
    runs-on: ubuntu-latest

    steps:
      # 🛎️ سحب الكود من المستودع
      - name: Checkout code
        uses: actions/checkout@v4

      # 🛠️ تثبيت Node.js
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'

      # 📦 تثبيت الاعتمادات
      - name: Install dependencies
        run: npm install

      # ✅ فحص قاعدة البيانات
      - name: Check DB
        run: npm run db:check

      # ♻️ إعادة تعيين قاعدة البيانات (اختياري)
      - name: Reset DB
        run: npm run db:reset

      # 🧪 تشغيل الاختبارات (لو أضفنا Jest لاحقًا)
      - name: Run tests
        run: |
          echo "No tests yet - add Jest or Mocha later"
