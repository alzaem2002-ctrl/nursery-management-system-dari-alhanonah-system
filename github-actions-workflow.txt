name: Node.js CI

on:
  push:
    branches: [ "main" ]
  pull_request:
    branches: [ "main" ]

jobs:
  build-and-test:
    runs-on: ubuntu-latest

    steps:
      # 🛎️ سحب الكود
      - name: Checkout code
        uses: actions/checkout@v4

      # 🛠️ تثبيت Node.js
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'

      # 📦 تثبيت الاعتمادات
      - name: Install dependencies
        run: npm install

      # ✅ فحص قاعدة البيانات
      - name: Check DB connection
        run: npm run db:check

      # ♻️ إعادة تعيين قاعدة البيانات
      - name: Reset DB
        run: npm run db:reset

      # 🚀 تشغيل السيرفر في الخلفية
      - name: Start server
        run: |
          nohup npm start &
          sleep 5

      # 🔎 اختبار /api/health
      - name: Test /api/health endpoint
        run: |
          curl -f http://localhost:3000/api/health | grep '"status":"ok"'

      # 🔎 اختبار /api/performance/health
      - name: Test /api/performance/health endpoint
        run: |
          curl -f http://localhost:3000/api/performance/health | grep '"status":"ok"'

      # ♻️ إعادة تعيين قاعدة البيانات أثناء تشغيل السيرفر
      - name: Reset DB during runtime
        run: npm run db:reset

      # 🔎 إعادة فحص /api/health بعد reset للتأكد أن السيرفر سليم
      - name: Re-test /api/health after DB reset
        run: |
          curl -f http://localhost:3000/api/health | grep '"status":"ok"'
